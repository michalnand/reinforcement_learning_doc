\begin{frame}
  
  \frametitle{\bf digital filters}

  \begin{itemize}

    \item low pass filter (1st order)
      \begin{align*}
        y_{n}&= (1 - a)y_{n-1} + a x_{n} \\
        a &= \frac{\Delta t}{RC + \Delta t}
      \end{align*}
  

    \item resonant filter (2nd order)
      \begin{align*}
        y_{n}&= a_{1}y_{n-1} + a_{2}y_{n-2} + b_{0}x_{n} + b_{1}x_{n-1} + b_{2}x_{n-2} 
      \end{align*}

  \end{itemize}

  
\end{frame}
